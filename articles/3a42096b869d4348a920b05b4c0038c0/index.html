[C#] IronPythonでC#側のオブジェクトを操作するだけのサンプル

ソースコード

環境は .NET Core 3.0 / C# 8.0

using System;
using IronPython.Hosting;

#nullable enable

class Program
{
    static void Main()
    {
        var sample = new Sample();

        var engine = Python.CreateEngine();
        var source = engine.CreateScriptSourceFromString("sample.Num = 100;");
        var scope = engine.CreateScope();
        scope.SetVariable("sample", sample);
        source.Execute(scope);

        Console.WriteLine(sample.Num);
    }
}

public class Sample
{
    public int Num { get; set; }
}
